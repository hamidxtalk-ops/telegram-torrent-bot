  preciseSearchModal.classList.remove('hidden');
  preciseMovieName.value = '';
  preciseMovieYear.value = '';
  preciseMovieName.focus();
});

preciseModalClose?.addEventListener('click', () => {
  preciseSearchModal.classList.add('hidden');
});

preciseSearchModal?.addEventListener('click', (e) => {
  if (e.target === preciseSearchModal) {
    preciseSearchModal.classList.add('hidden');
  }
});
